

MAN_add_or_modify_state_shintoism = {
	IF = {
		limit = {
			NOT = { has_dynamic_modifier = { modifier = MAN_state_shintoism_dm } }
		}
		custom_effect_tooltip = MAN_add_state_shinotism_modifier_tt
		hidden_effect = { add_dynamic_modifier = { modifier = MAN_state_shintoism_dm } }
	}
	ELSE = {
		custom_effect_tooltip = MAN_modify_state_shinotism_modifier_tt
	}
}
